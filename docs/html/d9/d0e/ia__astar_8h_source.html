100 
101  // Verifica se a origem e o destino não estão disponiveis.
102  if (!isUnBlocked(src,lvl) or !isUnBlocked(appll, lvl))
103  throw std::runtime_error("Source or the destination is blocked\n");
104 
105  // Recebe o mapa do nível atual.
106  Maps map = lvl.get_current_level();
107 
108  // Cria uma lista fechada e inicializa com falso, de forma que nenhuma célula tenha sido inicializada
109  // ainda.
110  bool closedList[map.return_y()][map.return_x()];
111  memset(closedList, false, sizeof (closedList));
112 
113 
114  m_cellDetails.resize(map.return_x());
115 
116  int i, j;
117 
118 
119  for (i = 0; i < map.return_x(); ++i)
120  {

123  {

128 
129  }

131 
132  // Inicializa os parâmetros da A*.
133  i = src.x, j = src.y;
134  m_cellDetails[i][j].f = 0;
135  m_cellDetails[i][j].g = 0;
136  m_cellDetails[i][j].h = 0;

138  m_cellDetails[i][j].parent = std::make_pair(i, j);
139 
140  // Cria uma lista aberta para armazenar os valores de f, g e h, de forma que f = g + h.
141  std::set<pPair> openList;
142 
143  // Coloca a primeira posição na lista aberta e configura para o f ser 0.
144  openList.insert(std::make_pair(0.0, std::make_pair (i, j)));
145 
146  while (!openList.empty())
147  {
148  pPair p = *openList.begin();
149 
150  // Remove esta posição da lista.
151  openList.erase(openList.begin());
152 
153  // Adiciona a posição a lista fechada.
154  i = p.second.first;
155  j = p.second.second;
156  closedList[i][j] = true;
157

159 
160  // Para cada uma das 4 posições verifica a partir da posição que se encontra.
161  Point points[4] = {Point(i, j+1), Point(i, j-1), Point(i+1, j), Point(i-1, j)};
162  uint c;
163  for(c = 0; c < 4; c++)
164  {
165  Point& current_point = points[c];
166

168  if (isValid(current_point, lvl))

170  // Se o destino for válido é igual ao sucessor atual.
171  if (isDestination(current_point, app))
172  {
173  // Configura o 'parent' da posição de destino.
174  m_cellDetails[current_point.x][current_point.y].parent = std::make_pair(i,j);
175  m_cellDetails[current_point.x][current_point.y].HasParent = true;
176  tracePath(app);
177  return true;
178  }
179  // Se o sucessor já está na lista fechada ou está bloqueada então ignore. Caso contrário siga.
180  else if (closedList[current_point.x][current_point.y] == false and
181  isUnBlocked(current_point, lvl) == true)
182  {
183  gNew = m_cellDetails[i][j].g + 1;
184  hNew = calculateHValue(current_point, app);
185  fNew = gNew + hNew;
186 
187  // Se não está na lista aberta, adicione-a. Faça a posição atual ser a posição do parent.
188  // Grava os valores de f, g e h da posição.
189  if (m_cellDetails[current_point.x][current_point.y].f == UINT_MAX or
190  m_cellDetails[current_point.x][current_point.y].f > fNew)
191  {
192  openList.insert( std::make_pair(fNew,
193  std::make_pair(current_point.x, current_point.y)));
194 
195  // Atualiza as informações da célula (bloco).
196  m_cellDetails[current_point.x][current_point.y].f = fNew;
197  m_cellDetails[current_point.x][current_point.y].g = gNew;
198  m_cellDetails[current_point.x][current_point.y].h = hNew;
199  m_cellDetails[current_point.x][current_point.y].parent = std::make_pair(i,j);
200  m_cellDetails[current_point.x][current_point.y].HasParent = true;
201  }
202  }
203  }
204  }
205  }
